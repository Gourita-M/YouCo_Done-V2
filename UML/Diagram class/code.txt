  #email : string
  #email_verified_at

}

  -id : int
  -name : string
  -city : string
  -cuisine_type : string
  -adress : string
  -capacity : int
  -openhours : float
  -closehours : float
  +addRestaurant()
  +removeRestaurant()
  +editRestaurant()
}

class images {
  -id : int
  -restuarant_image : string
}

class menu {
  -id : int
}

class menuplats {
  -id : int
}
class plat {
  -id : int
  -content : string
  -prix_uni : int
}

class favorites {
  -id : int

}

class Role {
  -id : int

}
